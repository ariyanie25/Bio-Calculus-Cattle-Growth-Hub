                        2. Mathematical Modeling
                    </h3>
                    <p class="text-sm text-slate-600 leading-relaxed mb-4">
                        Animal growth is classically modeled using the <strong>Logistic Growth Curve</strong>. The mass of the Kedah-Kelantan bull at age $t$ months is defined by the function:
                    </p>

                    <!-- LaTeX Display math -->
                    <div class="math-container bg-slate-50 p-4 rounded-xl mb-4 border border-slate-100 overflow-x-auto">
                        $$f(t) = \frac{L}{1 + a \cdot e^{-k \cdot t}}$$
                    </div>

                    <p class="text-sm text-slate-600 leading-relaxed mb-4">
                        To calculate the instantaneous rate of weight gain (change in mass over time), we must find the derivative $f'(t)$ using the <strong>Quotient Rule</strong> and <strong>Chain Rule</strong>:
                    </p>

                    <div class="math-container bg-slate-50 p-4 rounded-xl mb-4 border border-slate-100 overflow-x-auto">
                        $$f'(t) = \frac{L \cdot a \cdot k \cdot e^{-k \cdot t}}{(1 + a \cdot e^{-k \cdot t})^2}$$
                    </div>

                    <p class="text-xs text-slate-500 italic">
                        Where: <br>
                        • $L$ represents the mature asymptotic weight of the KK bull (kg).<br>
                        • $k$ represents the specific growth velocity constant ($month^{-1}$).<br>
                        • $a$ is a dimensionless scaling parameter related to initial birth weight.<br>
                        • $t$ is the age of the animal in months.
                    </p>
                </div>

    <script>
        // Parameters Object for different scenarios
        // Grounded in realistic MARDI-Kedah Kelantan Bull performance datasets
        const scenarios = {
            grass: {
                L: 280,   // Asymptotic mature weight on grass pasture in kg
                k: 0.14,  // Slower growth coefficient on pasture
                a: 13.0,  // Birth parameter (results in ~20kg birth weight)
                name: 'Standard Grazing System',
                desc: 'Brachiaria Decumbens Pasturing with mineral block supplementation.'
            },
            feedlot: {
                L: 340,   // Higher asymptotic weight due to heavy carbohydrate feedlot diet
                k: 0.19,  // Rapid pubertal growth coefficient
                a: 16.0,  // Selected stock baseline birth parameter (results in ~20kg birth weight)
                name: 'MARDI High-Performance Feedlot System',
                desc: 'Agricultural by-product concentrate feeding (Palm Kernel Cake base).'
            }
        };

        let currentScenarioKey = 'feedlot';
        let chart = null;

        // Mathematical growth functions
        function weightFunction(t, L, k, a) {
            return L / (1 + a * Math.exp(-k * t));
        }

        function derivativeFunction(t, L, k, a) {
            const expTerm = Math.exp(-k * t);
            const numerator = L * a * k * expTerm;
            const denominator = Math.pow(1 + a * expTerm, 2);
            return numerator / denominator;
        }

        // Get appropriate veterinary husbandry advice for KK cattle
        function getHusbandryAdvice(t, weight, monthlyRate, adg) {
            if (t < 3) {
                return {
                    title: "Neonatal/Calf Raising Phase",
                    text: `Age ${t} months. Birth/Weaning transition. Weight is ${weight.toFixed(1)} kg. High focus must be placed on mother's milk production (colostrum) and creep feed initiation. Daily gain is around ${Math.round(adg)}g/day.`
                };
            } else if (t >= 3 && t < 9) {
                return {
                    title: "Weaning & Rumen Development Phase",
                    text: `Age ${t} months. The calf is adjusting to high-forage diets. Rumen papillae are developing. Maintain good pasture quality with 12% Crude Protein. Average Daily Gain is growing steady at ${Math.round(adg)}g/day.`
                };
            } else if (t >= 9 && t < 15) {
                // Point of Inflection/Maximum growth rates usually around 11-14 months
                const maxGrowthAlert = (currentScenarioKey === 'feedlot' && t === 12) ? ' This is the inflection peak under current feedlot conditions!' : '';
                return {
                    title: "Pubertal Frame Acceleration Phase",
                    text: `Age ${t} months. Peak daily muscle accumulation phase! Weight is ${weight.toFixed(1)} kg.${maxGrowthAlert} Derivative is ${monthlyRate.toFixed(1)} kg/month (${Math.round(adg)}g/day). To capitalize on this high feed efficiency, provide high-quality protein-energy concentrates (e.g., PKC + Sago Meal).`
                };
            } else if (t >= 15 && t < 22) {
                return {
                    title: "Subcutaneous Fat Deposition (Finishing) Phase",
                    text: `Age ${t} months. Growth velocity has peaked and is now decelerating ($f''(t) < 0$). Feed conversion efficiency is declining. Animal is transitioning from muscle synthesis to fat deposition. Optimal economic slaughter window is nearing (target: 18-24 months).`
                };
            } else {
                return {
                    title: "Mature Stock/Maintenance Phase",
                    text: `Age ${t} months. The growth curve has plateaued close to its asymptote of ${scenarios[currentScenarioKey].L} kg. The derivative is very low (${monthlyRate.toFixed(2)} kg/month). Keeping feedlot steers past this age causes serious financial deficits due to very poor Feed Conversion Ratios (FCR).`
                };
            }
        }

        // Dynamic Update UI
        function updateCalculations(month) {
            const sc = scenarios[currentScenarioKey];
            const L = sc.L;
            const k = sc.k;
            const a = sc.a;

            // Compute values
            const weight = weightFunction(month, L, k, a);
            const rate = derivativeFunction(month, L, k, a);
            const adg = (rate * 1000) / 30.0; // convert kg/month to g/day

            // Update DOM text
            document.getElementById('outWeight').innerText = weight.toFixed(1);
            document.getElementById('outRate').innerText = (rate >= 0 ? '+' : '') + rate.toFixed(1);
            document.getElementById('outADG').innerText = Math.round(adg);

            // Update Math Steps
            const expVal = Math.exp(-k * month);
            document.getElementById('stepInput').innerText = `t = ${month} months (Selected)`;
            document.getElementById('stepExp').innerText = `e^(-${k} * ${month}) = ${expVal.toFixed(4)}`;
            document.getElementById('stepFunc').innerText = `f(${month}) = ${L} / (1 + ${a} * ${expVal.toFixed(4)}) = ${weight.toFixed(2)} kg`;
            
            const num = L * a * k * expVal;
            const den = Math.pow(1 + a * expVal, 2);
            document.getElementById('stepDeriv').innerText = `f'(${month}) = (${num.toFixed(2)}) / (${den.toFixed(4)}) = ${rate.toFixed(2)} kg/month`;

            // Update Veterinary Guidance
            const advice = getHusbandryAdvice(month, weight, rate, adg);
            document.getElementById('husbandryTitle').innerText = `Month ${month} - ${advice.title}`;
            document.getElementById('husbandryExplanation').innerText = advice.text;

            // Synchronize input elements
            document.getElementById('timelineSlider').value = month;
            document.getElementById('sliderValueDisplay').innerText = `Month ${month}`;

            // Draw indicator on chart
            updateChartIndicator(month);
            
            // Re-render LaTeX calculations dynamically
            if (window.MathJax && typeof MathJax.typesetPromise === "function") {
                MathJax.typesetPromise();
            }
        }

        function updateChartIndicator(month) {
            if (!chart) return;
            const sc = scenarios[currentScenarioKey];
            const weightVal = weightFunction(month, sc.L, sc.k, sc.a);
            const rateVal = derivativeFunction(month, sc.L, sc.k, sc.a);

            // Update the single indicator points
            const tempWeightArr = Array(37).fill(null);
            tempWeightArr[month] = weightVal;
            chart.data.datasets[2].data = tempWeightArr;

            const tempRateArr = Array(37).fill(null);
            tempRateArr[month] = rateVal;
            chart.data.datasets[3].data = tempRateArr;

            chart.update('none'); // silent update without resetting layout
        }

        // Sets feeding scenarios (Grazing pasture vs intensive commercial feedlot)
        function setScenario(key) {
            currentScenarioKey = key;
            const sc = scenarios[key];

            // Highlight chosen button
            const btnGrass = document.getElementById('btnScenarioGrass');
            const btnFeedlot = document.getElementById('btnScenarioFeedlot');

            if (key === 'grass') {
                btnGrass.className = "flex flex-col items-start p-4 rounded-xl border-2 transition text-left cursor-pointer focus:outline-none border-amber-500 bg-amber-50/50";
                btnFeedlot.className = "flex flex-col items-start p-4 rounded-xl border-2 transition text-left cursor-pointer focus:outline-none border-slate-200 hover:border-slate-300";
            } else {
                btnFeedlot.className = "flex flex-col items-start p-4 rounded-xl border-2 transition text-left cursor-pointer focus:outline-none border-emerald-500 bg-emerald-50/50";
                btnGrass.className = "flex flex-col items-start p-4 rounded-xl border-2 transition text-left cursor-pointer focus:outline-none border-slate-200 hover:border-slate-300";
            }

            // Update formula parameters display
            document.getElementById('paramL').innerText = `${sc.L} kg`;
            document.getElementById('paramK').innerText = sc.k;
            document.getElementById('paramA').innerText = sc.a.toFixed(1);

            // Re-populate chart coordinates
            const months = Array.from({length: 37}, (_, i) => i);
            const weightData = months.map(m => weightFunction(m, sc.L, sc.k, sc.a));
            const rateData = months.map(m => derivativeFunction(m, sc.L, sc.k, sc.a));

            chart.data.datasets[0].data = weightData;
            chart.data.datasets[1].data = rateData;
            
            // Adjust max constraints depending on scenario size
            chart.options.scales.y.max = key === 'feedlot' ? 400 : 320;
            chart.options.scales.y1.max = key === 'feedlot' ? 20 : 15;

            chart.update();

            // Rerun calculations on current active month
            const currentMonth = parseInt(document.getElementById('timelineSlider').value);
            updateCalculations(currentMonth);
        }

            const slider = document.getElementById('timelineSlider');

                updateCalculations(parseInt(e.target.value));
            });
        };
